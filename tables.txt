drop table Urls;
create table Urls
(
	UrlId int auto_increment,
	Url varchar(255),
	UrlComment varchar(255),
	Primary Key(UrlId)
);


drop table Attributes;
create table Attributes
(
	AttributeId int auto_increment,
	AttributeName varchar(255),
	RegexOrXpath varchar(10),
	Pattern varchar(255),
	Primary Key(AttributeId)
);
